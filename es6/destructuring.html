Destructuring

Object Destructuring

Destructuring is the syntax change that helps us do destructuring, as in it allows us to extract multiple values and bind them to multiple variables from Arrays and Objects with very simple syntax.

var foo = { bar: 'pony', baz: 3 }

var {bar, baz} = foo

console.log(bar)

console.log(baz)

// <- 'pony'

// <- 3

var foo = { bar: 'pony', baz: 3 }

var {bar: a, baz: b} = foo

console.log(a)

console.log(b)

// <- 'pony'

// <- 3

var foo = { bar: { deep: 'pony', dangerouslySetInnerHTML: 'lol' } }

var {bar: { deep, dangerouslySetInnerHTML: sure }} = foo

console.log(deep)

console.log(sure)

// <- 'pony'

// <- 'LOL'

var {foo} = {bar: 'baz'}

console.log(foo)

var {foo:{bar}} = {baz: 'ouch'}

// <- UNDEFINED

// <- EXCEPTION

Array Destructuring

we can do the same with arrays

var [a] = [10]

console.log(a)

// <- 10

var [,,a,b] = [1,2,3,4,5]

console.log(a)

// <- 3

As Default Function Arguments

This can be used as function arguments to have default values for the parameters.

function greet ({ age, name:greeting='she' }) {

console.log(${greeting} is ${age} years old.)

}

greet({ name: 'nico', age: 27 })

// <- 'nico is 27 years old'

greet({ age: 24 })

// <- 'SHE IS 24 YEARS OLD'
